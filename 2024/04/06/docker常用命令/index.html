Docker常用命令


docker pull centos:7 #拉取镜像

docker tag f5a6b296b8a2 docker.io/fengjay/nginx:v1.0.1 #docker 打标签 (重命名)

docker push fengjay/alpine:latest # docker 推送远程

docker rmi fengjay/alpine:latest #docker 删除镜像



docker rmi -f fengjay/alpine:latest #强制删除

docker ps #查看所有运行的容器
docker ps -a #查看所有的容器
 #运行容器
-i 表示启动一个可交互的容器,并持续打开标准输入
-t 表示使用终端关联到容器的标准输入输出上
-d 后台运行删除容器
-- name表示定义容器唯一名称
-rm 推出后删除

docker run [(-ti)option] --name 容器名称 centos:7 /bin/bash
#exit 退出容器并停止
#ctrl + p + q 退出容器不停止
docker exec -ti 容器名称或id /bin/bash #进入容器

docker start 容器名称或id #启动  
docker stop 容器名称或id #停止 
docker restart 容器名称或id #重启 

docker rm 容器名称或id #删除容器
docker logs 容器名称或id #查看日志

docker commit 容器名称或id docker.io/fengjay/mycentos  #容器转换为镜像

docker save id > 文件名.tar #镜像导出本地

docker load < 文件名.tar #镜像导入

##数据拷贝

docker cp 主机目录 容器id:容器目录

把目录里的东西拷贝的容器中

docker cp .容器id:容器目录

docker run -ti -d --rm -p 81:80 id  #docker 端口映射
